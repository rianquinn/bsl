Detailed Description

template<typename T>
class bsl::is_compound< T >

If the provided type is a compound type, provides the member constant value equal to true. Otherwise the member constant value is false.

#ifndef EXAMPLE_IS_COMPOUND_OVERVIEW_HPP
#define EXAMPLE_IS_COMPOUND_OVERVIEW_HPP
#include <bsl/is_compound.hpp>
#include <bsl/print.hpp>
#include "example_class_pod.hpp"
namespace bsl
{
inline void
example_is_compound_overview() noexcept
{
if (bsl::is_compound<example_class_pod>::value) {
bsl::print("success\n");
}
}
}
#endif
