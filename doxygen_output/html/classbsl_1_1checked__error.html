Detailed Description

This class provides the base type for all checked exceptions types defined by the AUTOSAR specification.

#include <bsl/checked_error.hpp>
#include <bsl/debug.hpp>
std::int32_t
main() noexcept
{
try {
throw bsl::checked_error{"checked error", "example"};
}
catch (bsl::checked_error const &e) {
bsl::print("{}\n", e);
}
catch (...) {
}
}
